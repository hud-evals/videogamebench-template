# Zelda: Link's Awakening DX - Preload
# Skips title screen, creates a new save file, skips Marin's opening dialogue,
# and exits the starting house so the LLM begins in the Koholint Island overworld.

# Wait for ROM to load
sleep 10

# ============================
# TITLE SCREEN → FILE SELECT
# START wakes the title, A presses advance past the animated intro
# ============================
press_key START
sleep 3
press_key A
sleep 2
press_key A
sleep 2
press_key A
sleep 3

# ============================
# FILE SELECT → NAME ENTRY → CONFIRM FILE
# START+START moves to file select, A selects empty slot → opens name entry,
# A types 1-2 chars (cursor lands wherever), START confirms name,
# A finalises back to file select
# ============================
press_key START
sleep 2
press_key START
sleep 2
press_key A
sleep 2
press_key A
sleep 2
press_key START
sleep 2
press_key A
sleep 5

# File now shows in player select — start it
press_key A
sleep 3
press_key A
sleep 3
press_key A
sleep 3
press_key START
sleep 8
